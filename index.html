[2013-10-30] Switch ContentEditable State (chrome)

ContentEditable
javascript:(function(){ var body = document.body, attr = 'contentEditable', txt = '[ContentEditable]'; if(body.getAttribute(attr)){ body.removeAttribute(attr); document.title = document.title.replace(txt, ''); }else{ body.setAttribute(attr, true); body.setAttribute('spellcheck', false); document.title = txt + document.title; } })();
